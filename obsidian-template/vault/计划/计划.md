const {WaitLoading, Beautify, Habit, Task} = await cJS();

await WaitLoading.wait(dv, 'habit/habit.md');

let {startDate, endDate} = Habit.defaultPeriod(dv);

let habits = dv.pages('"habit" and -"habit/habit_history"')
	.where(habit => habit.target && habit.file.name != '走路' && habit.pointsPerClock != 0)
	.map(habit => {
	    return {
			habit: habit,
			periodInfo: Habit.habitInfoBetween(dv, habit, startDate, endDate),
		}
	})
	.sort(habit => habit.periodInfo.progress)
	.slice(0, 800)
	.map(habit => `${habit.habit.file.link}[\`${Beautify.num2(habit.periodInfo.clockCounts)}/${Beautify.num2(habit.periodInfo.finalTarget)}\`](habit.md)`);

dv.paragraph('>[!min] \n' + habits.join(" ") + ' [[心理健康]] [[身体健康]] [[睡眠]] [[世界观]] [[科研]] [[饮食]] [[动画]] [[奖励]] [[礼物]] [[工作]]');

// await dv.view('misc/dataview-scripts/calendar', {"pages": dv.pages(`"${dv.current().file.folder}"`)});
Task.dv = dv;
Task.renderTasks(dv.pages(`"${dv.current().file.folder}"`), false);
