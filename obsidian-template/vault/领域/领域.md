# 领域
```dataviewjs
const {Files,WaitLoading} = await cJS();
await WaitLoading.wait();

let areas = Files.getFolderNotes('领域');

dv.table([""], areas.map(f => [dv.page(f).file.link]));
```

```datacorejsx
const {Files} = await cJS();

return function View() {
let areas = Files.getFolderNotes('领域');

const tableData = areas.map(page => ({
	displayLink: dc.fileLink(page)
}));
const COLUMNS = [{
	value: (row) => row.displayLink
}];

return <dc.Table rows={tableData} columns={COLUMNS} />;
}
```
